132  inline double compute_test_stat(double estimate,
133  std::string method,
134  double n_eff,
135  const std::vector<double>& x,
136  const std::vector<double>& y,
137  const std::vector<double>& weights)
138  {
139  // prevent overflow in atanh
140  if (estimate == 1.0)
141  estimate = 1 - 1e-12;
142  if (estimate == -1.0)
143  estimate = 1e-12;
144 
145  double stat;
146  if (methods::is_hoeffding(method)) {
147  stat = estimate / 30.0 + 1.0 / (36.0 * n_eff);
148  } else if (methods::is_kendall(method)) {
149  stat = estimate * impl::ktau_stat_adjust(x, y, weights);
150  } else if (methods::is_pearson(method)) {
151  stat = boost::math::atanh(estimate) * std::sqrt(n_eff - 3);
152  } else if (methods::is_spearman(method)) {
153  stat = boost::math::atanh(estimate) * std::sqrt((n_eff - 3) / 1.06);
154  } else if (methods::is_blomqvist(method)) {
155  stat = boost::math::atanh(estimate) * std::sqrt(n_eff);
156  } else {
157  throw std::runtime_error("method not implemented.");
158  }
159 
160  return stat;
161  }

163  inline double compute_p_value(double statistic,
164  std::string method,
165  std::string alternative,
166  double n_eff = 0.0)
167  {
168  double p_value;
169  if (methods::is_hoeffding(method)) {
170  if (n_eff == 0.0)
171  throw std::runtime_error("must provide n_eff for method 'hoeffd'.");
172  if (alternative != "two-sided")
173  throw std::runtime_error("only two-sided test available for Hoeffding's D.");
174  p_value = impl::phoeffb(statistic, n_eff);
175  } else {
176  boost::math::normal norm_dist(0, 1);
177  if (alternative == "two-sided") {
178  p_value = 2 * boost::math::cdf(norm_dist, -std::abs(statistic));
179  } else if (alternative == "less") {
180  p_value = boost::math::cdf(norm_dist, statistic);
181  } else if (alternative == "greater") {
182  p_value = 1 - boost::math::cdf(norm_dist, statistic);
183  } else {
184  throw std::runtime_error("alternative not implemented.");
185  }
186  }
187 
188  return p_value;
189  }
